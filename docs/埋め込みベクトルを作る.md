import argparse
import hashlib
import shutil
from dataclasses import dataclass
from pathlib import Path
from typing import Literal, List, Any, Optional
from collections import Counter

import numpy as np
import torch
from sentence_transformers import SentenceTransformer

import zvec


# ===== コレクション設定 =====
DEFAULT_COLLECTION_PATH = "./zvec_md_collection_ruri"
COLLECTION_NAME = "md_docs"
VECTOR_FIELD = "embedding"

# スカラー fields
FIELD_SOURCE_PATH = "source_path"
FIELD_TITLE = "title"
FIELD_CHUNK_ID = "chunk_id"
FIELD_CONTENT = "content"


# ===== Ruri v3-310m 埋め込み =====
RuriMode = Literal["semantic", "topic", "query", "document"]

_RURI_MODEL: Optional[SentenceTransformer] = None
_RURI_DEVICE: Optional[str] = None


# --- ヘルパー関数: Status / dict 両対応 ---
def is_ok(st: Any) -> bool:
    if hasattr(st, "ok") and callable(getattr(st, "ok")):
        return bool(st.ok())
    if isinstance(st, dict) and "code" in st:
        return st["code"] == 0
    return False


def code_of(st: Any) -> Any:
    if hasattr(st, "code") and callable(getattr(st, "code")):
        return st.code()
    if isinstance(st, dict) and "code" in st:
        return st["code"]
    return None


def msg_of(st: Any) -> str:
    if hasattr(st, "message") and callable(getattr(st, "message")):
        return str(st.message())
    if isinstance(st, dict) and "message" in st:
        return str(st["message"])
    return ""


def is_already_exists(st: Any) -> bool:
    """
    Zvec の ALREADY_EXISTS を成功(=INFO)寄りに扱うための判定。
    """
    c = code_of(st)
    # StatusCode enum の場合
    if c == zvec.StatusCode.ALREADY_EXISTS:
        return True
    # dict の場合: code が int(2) で来る可能性にも対応
    if c == getattr(zvec.StatusCode.ALREADY_EXISTS, "value", None):
        return True
    return False


def report_insert_statuses(statuses: Any) -> None:
    """
    ポリシー:
      - OK: is_ok() == True
      - INFO: ALREADY_EXISTS（既に存在する場合）
      - NG: それ以外の失敗
    """
    if not isinstance(statuses, list):
        c = code_of(statuses)
        m = msg_of(statuses)
        if is_ok(statuses):
            print(f"Insert result: OK (code={c})")
        elif is_already_exists(statuses):
            print(f"Insert result: INFO already exists (code={c})")
            if m:
                print("  message:", m)
        else:
            print(f"Insert result: NG (code={c})")
            if m:
                print("  message:", m)
        return

    total = len(statuses)
    ok_list = [s for s in statuses if is_ok(s)]
    info_list = [s for s in statuses if (not is_ok(s)) and is_already_exists(s)]
    ng_list = [s for s in statuses if (not is_ok(s)) and (not is_already_exists(s))]

    print(f"Inserted {total} docs (ok={len(ok_list)}, info={len(info_list)}, ng={len(ng_list)})")

    # INFO 集計（必要なら表示）
    if info_list:
        info_codes = [code_of(s) for s in info_list]
        print("INFO code summary:", Counter(info_codes))
        sample = info_list[0]
        print("Sample INFO code:", code_of(sample))
        print("Sample INFO message:", msg_of(sample))

    # NG がある場合のみ詳細表示
    if ng_list:
        ng_codes = [code_of(s) for s in ng_list]
        print("NG code summary:", Counter(ng_codes))
        sample = ng_list[0]
        print("Sample NG code:", code_of(sample))
        print("Sample NG message:", msg_of(sample))


def _get_ruri_v3_310m(device: Optional[str] = None) -> SentenceTransformer:
    """
    cl-nagoya/ruri-v3-310m を（必要なら）ロードして返す。
    """
    global _RURI_MODEL, _RURI_DEVICE

    if device is None:
        device = "cuda" if torch.cuda.is_available() else "cpu"

    if _RURI_MODEL is not None and _RURI_DEVICE == device:
        return _RURI_MODEL

    _RURI_MODEL = SentenceTransformer("cl-nagoya/ruri-v3-310m", device=device)
    _RURI_DEVICE = device
    return _RURI_MODEL


def embed_with_ruri_v3_310m(
    texts: List[str],
    mode: RuriMode,
    *,
    device: Optional[str] = None,
    batch_size: int = 64,
    normalize: bool = True,
    show_progress_bar: bool = True,
) -> np.ndarray:
    """
    Ruri v3-310m で埋め込みを作る（モデルカードの 1+3 プレフィックス方式）
      semantic: ""（空文字）
      topic:    "トピック: "
      query:    "検索クエリ: "
      document: "検索文書: "
    戻り値:
      shape = (len(texts), dim) の float32 numpy 配列
    """
    prefix_map = {
        "semantic": "",
        "topic": "トピック: ",
        "query": "検索クエリ: ",
        "document": "検索文書: ",
    }
    prefix = prefix_map[mode]
    inputs = [prefix + t for t in texts]

    model = _get_ruri_v3_310m(device=device)
    emb = model.encode(
        inputs,
        batch_size=batch_size,
        convert_to_numpy=True,
        normalize_embeddings=normalize,
        show_progress_bar=show_progress_bar,
    )
    return emb.astype("float32")


# ===== Markdown 読み込み / 分割 =====
@dataclass(frozen=True)
class Chunk:
    source_path: str
    title: str
    chunk_id: int
    content: str


def extract_title(md_text: str, fallback: str) -> str:
    """
    Markdown先頭の '# ' 見出しをタイトルとして抽出（なければ fallback）
    """
    for line in md_text.splitlines()[:80]:
        s = line.strip()
        if s.startswith("# "):
            return s[2:].strip() or fallback
    return fallback


def iter_chunks(text: str, chunk_chars: int, overlap_chars: int) -> List[str]:
    """
    文字数ベースの単純な分割。
      - chunk_chars <= 0 の場合は分割しない（全文を1つだけ返す）
      - overlap_chars は次チャンクへ持ち越す文字数
    """
    text = text or ""
    if chunk_chars <= 0:
        return [text]

    if overlap_chars < 0:
        overlap_chars = 0
    if overlap_chars >= chunk_chars:
        overlap_chars = max(0, chunk_chars // 4)

    out: List[str] = []
    start = 0
    n = len(text)
    while start < n:
        end = min(n, start + chunk_chars)
        out.append(text[start:end])
        if end == n:
            break
        start = max(0, end - overlap_chars)
    return out


def load_markdown(docs_dir: Path) -> List[Path]:
    md_files = sorted(docs_dir.glob("*.md"))
    return md_files


def make_chunks(md_files: List[Path], chunk_chars: int, overlap_chars: int) -> List[Chunk]:
    chunks: List[Chunk] = []
    for f in md_files:
        text = f.read_text(encoding="utf-8")
        title = extract_title(text, fallback=f.stem)
        parts = iter_chunks(text, chunk_chars=chunk_chars, overlap_chars=overlap_chars)
        for i, part in enumerate(parts):
            part = part.strip()
            if not part:
                continue
            chunks.append(
                Chunk(
                    source_path=str(f),
                    title=title,
                    chunk_id=i,
                    content=part,
                )
            )
    return chunks


def stable_doc_id(source_path: str, chunk_id: int) -> str:
    """
    Zvecのid制約（正規表現）に確実に通るよう、ASCIIの [a-z0-9_] のみにする。
    - doc_ で英字開始
    - sha1は [0-9a-f]
    """
    h = hashlib.sha1(f"{source_path}::{chunk_id}".encode("utf-8")).hexdigest()  # 40 chars, 0-9a-f
    return f"doc_{h}_{chunk_id:04d}"


# ===== Zvec コレクション =====
def build_schema(embedding_dim: int) -> zvec.CollectionSchema:
    """
    Zvec のスキーマ（fields + vectors）
    """
    return zvec.CollectionSchema(
        name=COLLECTION_NAME,
        fields=[
            zvec.FieldSchema(FIELD_SOURCE_PATH, zvec.DataType.STRING),
            zvec.FieldSchema(FIELD_TITLE, zvec.DataType.STRING, nullable=True),
            zvec.FieldSchema(FIELD_CHUNK_ID, zvec.DataType.INT32),
            zvec.FieldSchema(FIELD_CONTENT, zvec.DataType.STRING),
        ],
        vectors=[
            zvec.VectorSchema(
                name=VECTOR_FIELD,
                data_type=zvec.DataType.VECTOR_FP32,
                dimension=embedding_dim,
                index_param=zvec.HnswIndexParam(metric_type=zvec.MetricType.COSINE),
            )
        ],
    )


def open_or_create_collection(path: Path, schema: zvec.CollectionSchema) -> zvec.Collection:
    """
    既に存在すれば open()、無ければ create_and_open()。
    """
    if path.exists():
        coll = zvec.open(str(path))
        # 念のため次元一致チェック（異なると insert/query で困る）
        v = coll.schema.vector(VECTOR_FIELD)
        if v is None:
            raise RuntimeError(f"Vector field '{VECTOR_FIELD}' not found in existing collection schema.")
        if int(v.dimension) != int(schema.vector(VECTOR_FIELD).dimension):  # type: ignore[union-attr]
            raise RuntimeError(
                f"Embedding dimension mismatch: existing={v.dimension}, requested={schema.vector(VECTOR_FIELD).dimension}.\n"
                f"Tip: --recreate で作り直してください。"
            )
        return coll

    return zvec.create_and_open(path=str(path), schema=schema)


# ===== メイン =====
def main() -> None:
    parser = argparse.ArgumentParser()
    parser.add_argument("--docs_dir", type=str, default="./docs")
    parser.add_argument("--collection_path", type=str, default=DEFAULT_COLLECTION_PATH)
    parser.add_argument("--query", type=str, default=None)
    parser.add_argument("--topk", type=int, default=10)
    parser.add_argument("--device", type=str, default=None, help="例: cpu / cuda")

    # 分割設定
    # chunk_chars=0 にすれば分割しない
    parser.add_argument("--chunk_chars", type=int, default=300)
    parser.add_argument("--overlap_chars", type=int, default=50)

    # 既存コレクションを削除して作り直す
    parser.add_argument("--recreate", action="store_true")

    args = parser.parse_args()

    docs_dir = Path(args.docs_dir)
    if not docs_dir.exists():
        raise FileNotFoundError(f"docs_dir not found: {docs_dir}")

    collection_path = Path(args.collection_path)
    if args.recreate and collection_path.exists():
        shutil.rmtree(collection_path)

    # 1) Ruriモデルをロードして次元取得（スキーマ次元と一致必須）
    ruri = _get_ruri_v3_310m(device=args.device)
    embedding_dim = int(ruri.get_sentence_embedding_dimension())
    if embedding_dim <= 0:
        raise RuntimeError("Failed to get embedding dimension from cl-nagoya/ruri-v3-310m.")

    schema = build_schema(embedding_dim=embedding_dim)
    collection = open_or_create_collection(collection_path, schema=schema)

    # 2) Markdownを読み込み
    md_files = load_markdown(docs_dir)
    if not md_files:
        raise RuntimeError(f"No markdown files found in: {docs_dir}")
    # if len(md_files) < 10:
    #     print(f"Warning: markdown files found = {len(md_files)} (expected 10). Using what we have.")

    # 3) チャンク化（必要なら）
    chunks = make_chunks(md_files, chunk_chars=args.chunk_chars, overlap_chars=args.overlap_chars)
    if not chunks:
        raise RuntimeError("All markdown files were empty after preprocessing.")

    # 4) 文書埋め込み（検索対象文書は '検索文書: '）
    doc_texts = [c.content for c in chunks]
    doc_emb = embed_with_ruri_v3_310m(
        doc_texts,
        mode="document",
        device=args.device,
        show_progress_bar=True,
        normalize=True,
    )

    # 5) Zvec に insert()
    docs: List[zvec.Doc] = []
    for c, emb in zip(chunks, doc_emb):
        docs.append(
            zvec.Doc(
                id=stable_doc_id(c.source_path, c.chunk_id),
                vectors={VECTOR_FIELD: emb.tolist()},
                fields={
                    FIELD_SOURCE_PATH: c.source_path,
                    FIELD_TITLE: c.title,
                    FIELD_CHUNK_ID: int(c.chunk_id),
                    FIELD_CONTENT: c.content,
                },
            )
        )

    statuses = collection.insert(docs)
    # statuses は Status か list[Status]

    # 結果の表示
    report_insert_statuses(statuses)
    
    # 6) まとめて投入したら optimize()（推奨）
    collection.optimize()
    collection.flush()

    # 7) 検索クエリ埋め込み（検索クエリは '検索クエリ: '）
    query = args.query
    if not query:
        query = input("Query text> ").strip()
    if not query:
        raise RuntimeError("Empty query.")

    q_emb = embed_with_ruri_v3_310m(
        [query],
        mode="query",
        device=args.device,
        show_progress_bar=False,
        normalize=True,
    )[0].tolist()

    results = collection.query(
        vectors=zvec.VectorQuery(VECTOR_FIELD, vector=q_emb),
        topk=args.topk,
        output_fields=[FIELD_SOURCE_PATH, FIELD_TITLE, FIELD_CHUNK_ID, FIELD_CONTENT],
        include_vector=False,
    )
    
    # スコア降順に並び替え（Noneは最下位扱い）
    results_sorted = sorted(
        results,
        key=lambda d: float(d.score) if d.score is not None else float("-inf"),
        reverse=True,
    )

    print("\n=== Search Results ===")
    for rank, doc in enumerate(results_sorted, start=1):
        title = doc.field(FIELD_TITLE)
        src = doc.field(FIELD_SOURCE_PATH)
        cid = doc.field(FIELD_CHUNK_ID)
        content = doc.field(FIELD_CONTENT) or ""
        snippet = str(content).replace("\n", " ")[:180]
        score = doc.score if doc.score is not None else float("nan")
        print(f"[{rank}] score={score:.4f}  title={title}  chunk={cid}  src={src}")
        print(f"     {snippet}\n")


if __name__ == "__main__":
    main()
